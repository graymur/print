This is the description of IncrementinbBox script, which allows to input fields with ability to increase and decrease it's value with mouse clicks.
Script was tested in IE 6, FF 1.5, Opera 9.

Simplest example:

(new IncrementingBox).appendTo('ib1');
Script allows to set default value, the increment value, maximum and minimun values:


In this example default value is 1.5, increment value (the value which will be added or subtracted) is 0.1, minimum value is 0, maximum - 1.5
var IBox = new IncrementingBox;
IBox.default_value   = 0.5;
IBox.increment_value = 0.1;
IBox.min             = 0;
IBox.max             = 1.5;

IBox.appendTo('ib2');
Script also allows to assing CSS classes to input field, container of array images, set event handlers for input field and define a function, that will be executed every time the value is changed by clicking arrows or typing directly into the input field.


In this example CSS classes for input and contaier are assigned, handlers for onMouseOver and onMouseOut event for input field are set, and also, the function that is executed on every change in input field is defined, 
IBox.onIncrementCallback = 'incrementCallback'
.
In this example this function takes current value of incrementing box, and puts its squared value into the next input field. 
var IBox = new IncrementingBox;
IBox.default_value   = 1;
IBox.increment_value = 0.2;
IBox.min             = 0;
IBox.max             = 5;

IBox.input_class_name     = 'ibox_input';
IBox.container_class_name = 'ibox_container';

IBox.input_onmouseover = mouseOverHandler;
IBox.input_onmouseout  = mouseOutHandler;

IBox.onIncrementCallback = 'incrementCallback';

IBox.appendTo('ib3');

function mouseOverHandler(e)
{
    if (isIE) e = window.event;
    if (isMoz) e.srcElement = e.target;
    
    e.srcElement.style.backgroundColor = 'black';
    e.srcElement.style.color = 'white';
}

function mouseOutHandler(e)
{
    if (isIE) e = window.event;
    if (isMoz) e.srcElement = e.target;

    e.srcElement.style.backgroundColor = 'white';
    e.srcElement.style.color = 'black';
}

function incrementCallback()
{
    document.getElementById('in1').value = currentIBox.value * currentIBox.value;
}
Another usefull method of the script is 
.replace(element_id)
, it might be used when you need to replace regular input value with incrementing. For example, we following HTML: 
<div>
  <input type="text" id="in2" />
  <input type="button" value="Submit">
</div>
and we need to have incrementing box in the place of regular. In this case, you can use 
.replace(element_id)
method: 
var IBox = new IncrementingBox;
IBox.replace('in2');
